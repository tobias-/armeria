38  /**
39   * Collects the metric data and stores it into the {@link MeterRegistry}.
40   */
41  public final class RequestMetricSupport {
42  
43      private static final AttributeKey<RequestMetrics> ATTR_REQUEST_METRICS =
44              AttributeKey.valueOf(RequestMetricSupport.class, "REQUEST_METRICS");
45  
46      private static final RequestMetrics PLACEHOLDER = new RequestMetricsPlaceholder();
47  
48      public static void setup(RequestContext ctx, MeterIdPrefixFunction meterIdPrefixFunction) {
49          if (ctx.hasAttr(ATTR_REQUEST_METRICS)) {
50              return;
51          }
52  
53          // Set the attribute to the placeholder so that calling this method again has no effect.
54          // The attribute will be set to the real one later in onRequestStart()
55          ctx.attr(ATTR_REQUEST_METRICS).set(PLACEHOLDER);
56  
57          ctx.log().addListener(log -> onRequestStart(log, meterIdPrefixFunction),
58                                RequestLogAvailability.REQUEST_HEADERS,
59                                RequestLogAvailability.REQUEST_CONTENT);
60          ctx.log().addListener(RequestMetricSupport::onRequestEnd,
61                                RequestLogAvailability.REQUEST_END);
62          ctx.log().addListener(RequestMetricSupport::onResponse,
63                                RequestLogAvailability.COMPLETE);
64      }
65  
66      private static void onRequestStart(RequestLog log, MeterIdPrefixFunction meterIdPrefixFunction) {
67          final RequestContext ctx = log.context();
68          final MeterRegistry registry = ctx.meterRegistry();
69          final MeterIdPrefix idPrefix = meterIdPrefixFunction.apply(registry, log);
70          final RequestMetrics requestMetrics =  MicrometerUtil.register(
71                  registry, idPrefix, RequestMetrics.class, DefaultRequestMetrics::new);
72  
73          ctx.attr(ATTR_REQUEST_METRICS).set(requestMetrics);
74          requestMetrics.active().incrementAndGet();
75      }
76  
77      private static void onRequestEnd(RequestLog log) {
78          final RequestMetrics metrics = requestMetrics(log);
79          metrics.requestDuration().record(log.requestDurationNanos(), TimeUnit.NANOSECONDS);
80          metrics.requestLength().record(log.requestLength());
81          if (log.requestCause() != null) {
82              metrics.failure().increment();
83              metrics.active().decrementAndGet();
84          }
85      }
86  
87      private static void onResponse(RequestLog log) {
88          if (log.requestCause() != null) {
89              return;
90          }
91  
92          final RequestMetrics metrics = requestMetrics(log);
93          metrics.responseDuration().record(log.responseDurationNanos(), TimeUnit.NANOSECONDS);
94          metrics.responseLength().record(log.responseLength());
95          metrics.totalDuration().record(log.totalDurationNanos(), TimeUnit.NANOSECONDS);
96  
97          if (isSuccess(log)) {
98              metrics.success().increment();
99          } else {
100             metrics.failure().increment();
101         }
102 
103         metrics.active().decrementAndGet();
104     }
105 
106     private static boolean isSuccess(RequestLog log) {
107         if (log.responseCause() != null) {
108             return false;
109         }
110 
111         final int statusCode = log.statusCode();
112         if (statusCode < 100 || statusCode >= 400) {
113             return false;
114         }
115 
116         final Object responseContent = log.responseContent();
117         if (responseContent instanceof RpcResponse) {
118             return !((RpcResponse) responseContent).isCompletedExceptionally();
119         }
120 
121         return true;
122     }
123 
124     private static RequestMetrics requestMetrics(RequestLog log) {
125         return log.context().attr(ATTR_REQUEST_METRICS).get();
126     }
127 
128     private RequestMetricSupport() {}
129 
130     private interface RequestMetrics {
131         AtomicInteger active();
132 
133         Counter success();
134 
135         Counter failure();
136 
137         Timer requestDuration();
138 
139         DistributionSummary requestLength();
140 
141         Timer responseDuration();
142 
143         DistributionSummary responseLength();
144 
145         Timer totalDuration();
146     }
147 
148     private static final class DefaultRequestMetrics implements RequestMetrics {
149 
150         private final AtomicInteger active;
151         private final Counter success;
152         private final Counter failure;
153         private final Timer requestDuration;
154         private final DistributionSummary requestLength;
155         private final Timer responseDuration;
156         private final DistributionSummary responseLength;
157         private final Timer totalDuration;
158 
159         DefaultRequestMetrics(MeterRegistry parent, MeterIdPrefix idPrefix) {
160             active = parent.gauge(idPrefix.name("activeRequests"), idPrefix.tags(),
161                                   new AtomicInteger(), AtomicInteger::get);
162             final String requests = idPrefix.name("requests");
163             success = parent.counter(requests, idPrefix.tags("result", "success"));
164             failure = parent.counter(requests, idPrefix.tags("result", "failure"));
165 
166             requestDuration = timerWithDefaultQuantiles(
167                     parent, idPrefix.name("requestDuration"), idPrefix.tags());
168             requestLength = summaryWithDefaultQuantiles(
169                     parent, idPrefix.name("requestLength"), idPrefix.tags());
170             responseDuration = timerWithDefaultQuantiles(
171                     parent, idPrefix.name("responseDuration"), idPrefix.tags());
172             responseLength = summaryWithDefaultQuantiles(
173                     parent, idPrefix.name("responseLength"), idPrefix.tags());
174             totalDuration = timerWithDefaultQuantiles(
175                     parent, idPrefix.name("totalDuration"), idPrefix.tags());
176         }
177 
178         @Override
179         public AtomicInteger active() {
180             return active;
181         }
182 
183         @Override
184         public Counter success() {
185             return success;
186         }
187 
188         @Override
189         public Counter failure() {
190             return failure;
191         }
192 
193         @Override
194         public Timer requestDuration() {
195             return requestDuration;
196         }
197 
198         @Override
199         public DistributionSummary requestLength() {
200             return requestLength;
201         }
202 
203         @Override
204         public Timer responseDuration() {
205             return responseDuration;
206         }
207 
208         @Override
209         public DistributionSummary responseLength() {
210             return responseLength;
211         }
212 
213         @Override
214         public Timer totalDuration() {
215             return totalDuration;
216         }
217     }
218

259     }
