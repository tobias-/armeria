30  package com.linecorp.armeria.internal.logging;
31  
32  import static com.google.common.base.Preconditions.checkArgument;
33  
34  /**
35   * Sampler is responsible for deciding if a particular trace should be "sampled", i.e. whether the
36   * overhead of tracing will occur and/or if a trace will be reported to the collection tier.
37   *
38   * <p>Zipkin v1 uses before-the-fact sampling. This means that the decision to keep or drop the
39   * trace is made before any work is measured, or annotations are added. As such, the input parameter
40   * to zipkin v1 samplers is the trace ID (64-bit random number).
41   *
42   * <p>The instrumentation sampling decision happens once, at the root of the trace, and is
43   * propagated downstream. For this reason, the decision needn't be consistent based on trace ID.
44   *
45   * <p>Forked from brave-core.
46   */
47  // abstract for factory-method support on Java language level 7
48  public abstract class Sampler {
49  
50      static final Sampler ALWAYS_SAMPLE = new Sampler() {
51          @Override
52          public boolean isSampled() {
53              return true;
54          }
55  
56          @Override
57          public String toString() {
58              return "AlwaysSample";
59          }
60      };
61  
62      static final Sampler NEVER_SAMPLE = new Sampler() {
63          @Override
64          public boolean isSampled() {
65              return false;
66          }
67  
68          @Override
69          public String toString() {
70              return "NeverSample";
71          }
72      };
73  
74      /**
75       *  Returns true if a log should be recorded.
76       */
77      public abstract boolean isSampled();
78  
79      /**
80       * Returns a sampler, given a rate expressed as a percentage.
81       *
82       * <p>The sampler returned is good for low volumes of traffic (<100K requests), as it is precise.
83       * If you have high volumes of traffic, consider {@code BoundarySampler}.
84       *
85       * @param rate 0 for no sampling, 1 for full sampling, or in [0.01, 1] for random sampling.
86       */
87      public static Sampler create(float rate) {
88          if (rate == 1.0) {
89              return ALWAYS_SAMPLE;
90          } else if (rate == 0.0) {
91              return NEVER_SAMPLE;
92          }
93          checkArgument(rate >= 0.01f && rate < 1, "rate should be between 0.01 and 1: was %s", rate);
94          return new CountingSampler(rate);
95      }
96  
97      /**
98       * Returns a {@link Sampler} for full sampling.
99       */
100     public static Sampler always() {
101         return ALWAYS_SAMPLE;
102     }
103 }
