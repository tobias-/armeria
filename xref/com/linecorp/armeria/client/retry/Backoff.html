16  package com.linecorp.armeria.client.retry;
17  
18  import static com.google.common.base.Preconditions.checkArgument;
19  import static com.linecorp.armeria.client.retry.FixedBackoff.NO_DELAY;
20  import static java.util.Objects.requireNonNull;
21  
22  import java.util.Optional;
23  import java.util.Random;
24  import java.util.concurrent.ThreadLocalRandom;
25  import java.util.function.Supplier;
26  
27  /**
28   * Control back off between attempts in a single retry operation.
29   */
30  @FunctionalInterface
31  public interface Backoff {
32      /**
33       * Returns a {@link Backoff} that that will never wait between attempts.
34       */
35      static Backoff withoutDelay() {
36          return NO_DELAY;
37      }
38  
39      /**
40       * Returns a {@link Backoff} that waits a fixed delay between attempts.
41       */
42      static Backoff fixed(long delayMillis) {
43          return new FixedBackoff(delayMillis);
44      }
45  
46      /**
47       * Returns a {@link Backoff} that waits an exponentially-increasing amount of time between attempts.
48       */
49      static Backoff exponential(long initialDelayMillis, long maxDelayMillis) {
50          return exponential(initialDelayMillis, maxDelayMillis, 2.0);
51      }
52  
53      /**
54       * Returns a {@link Backoff} that waits an exponentially-increasing amount of time between attempts.
55       */
56      static Backoff exponential(long initialDelayMillis, long maxDelayMillis, double multiplier) {
57          return new ExponentialBackoff(initialDelayMillis, maxDelayMillis, multiplier);
58      }
59  
60      /**
61       * Returns a {@link Backoff} that computes backoff delay which is a random value between
62       * {@code minDelayMillis} and {@code maxDelayMillis} chosen by {@link ThreadLocalRandom}.
63       */
64      static Backoff random(long minDelayMillis, long maxDelayMillis) {
65          return random(minDelayMillis, maxDelayMillis, ThreadLocalRandom::current);
66      }
67  
68      /**
69       * Returns a {@link Backoff} that computes backoff delay which is a random value between
70       * {@code minDelayMillis} and {@code maxDelayMillis}.
71       */
72      static Backoff random(long minDelayMillis, long maxDelayMillis, Supplier<Random> randomSupplier) {
73          return new RandomBackoff(minDelayMillis, maxDelayMillis, randomSupplier);
74      }
75  
76      /**
77       * Creates a new {@link Backoff} that computes backoff delay using one of
78       * {@link #exponential(long, long, double)}, {@link #fixed(long)} and {@link #random(long, long)} chaining
79       * with {@link #withJitter(double, double)} and {@link #withMaxAttempts(int)} from the
80       * {@code specification}. This is the format for the {@code specification}:
81       * <ul>
82       *   <li>{@code exponential=[initialDelayMillis:maxDelayMillis:multiplier]} is for
83       *       {@link Backoff#exponential(long, long, double)} (multiplier will be 2.0 if it's omitted)</li>
84       *   <li>{@code fixed=[delayMillis]} is for {@link Backoff#fixed(long)}</li>
85       *   <li>{@code random=[minDelayMillis:maxDelayMillis]} is for {@link Backoff#random(long, long)}</li>
86       *   <li>{@code jitter=[minJitterRate:maxJitterRate]} is for {@link Backoff#withJitter(double, double)}
87       *       (if only one jitter value is specified, it will be used for {@link Backoff#withJitter(double)}</li>
88       *   <li>{@code maxAttempts=[maxAttempts]} is for {@link Backoff#withMaxAttempts(int)}</li>
89       * </ul>
90       * The order of options does not matter, and the {@code specification} needs at least one option.
91       * If you don't specify the base option exponential backoff will be used. If you only specify
92       * a base option, jitter and maxAttempts will be set by default values.
93       * These are a few examples:
94       * <ul>
95       *   <li>{@code exponential=200:10000:2.0,jitter=0.2} (default)</li>
96       *   <li>{@code exponential=200:10000,jitter=0.2,maxAttempts=50} (multiplier omitted)</li>
97       *   <li>{@code fixed=100,jitter=-0.5:0.2,maxAttempts=10} (fixed backoff with jitter variation)</li>
98       *   <li>{@code random=200:1000} (jitter and maxAttempts will be set by default values)</li>
99       * </ul>
100      *
101      * @param specification the specification used to create the {@link Backoff}
102      */
103     static Backoff of(String specification) {
104         BackoffSpec backoffSpec = BackoffSpec.parse(specification);
105         return backoffSpec.build();
106     }
107 
108     /**
109      * Returns the number of milliseconds to wait for before attempting a retry.
110      *
111      * @param numAttemptsSoFar the number of attempts made by a client so far, including the first attempt and
112      *                         its following retries.
113      *
114      * @return the number of milliseconds to wait for before attempting a retry,
115      *         or a negative value if no further retry has to be made.
116      *
117      * @throws IllegalArgumentException if {@code numAttemptsSoFar} is equal to or less than {@code 0}
118      */
119     long nextDelayMillis(int numAttemptsSoFar);
120 
121     /**
122      * Undecorates this {@link Backoff} to find the {@link Backoff} which is an instance of the specified
123      * {@code backoffType}.
124      *
125      * @param backoffType the type of the desired {@link Backoff}
126      * @return the {@link Backoff} which is an instance of {@code backoffType} if this {@link Backoff}
127      *         decorated such a {@link Backoff}. {@link Optional#empty()} otherwise.
128      */
129     default <T> Optional<T> as(Class<T> backoffType) {
130         requireNonNull(backoffType, "backoffType");
131         return backoffType.isInstance(this) ? Optional.of(backoffType.cast(this))
132                                             : Optional.empty();
133     }
134 
135     /**
136      * Returns a {@link Backoff} that adds a random jitter value to the original delay using
137      * <a href="https://www.awsarchitectureblog.com/2015/03/backoff.html">full jitter</a> strategy.
138      * The {@code jitterRate} is used to calculate the lower and upper bound of the ultimate delay.
139      * The lower bound will be {@code ((1 - jitterRate) * originalDelay)} and the upper bound will be
140      * {@code ((1 + jitterRate) * originalDelay)}. For example, if the delay returned by
141      * {@link Backoff#exponential(long, long)} is 1000 milliseconds and the provided jitter value is 0.3,
142      * the ultimate backoff delay will be chosen between 1000 * (1 - 0.3) and 1000 * (1 + 0.3)
143      * by {@link ThreadLocalRandom}. The rate value should be between 0.0 and 1.0.
144      *
145      * @param jitterRate the rate that used to calculate the lower and upper bound of the backoff delay
146      * @throws IllegalArgumentException if {@code jitterRate} is a negative value or greater than 1.0
147      */
148     default Backoff withJitter(double jitterRate) {
149         checkArgument(0.0 <= jitterRate && jitterRate <= 1.0,
150                       "jitterRate: %s (expected: >= 0.0 and <= 1.0)", jitterRate);
151         return withJitter(-jitterRate, jitterRate, ThreadLocalRandom::current);
152     }
153 
154     /**
155      * Returns a {@link Backoff} that adds a random jitter value to the original delay using
156      * <a href="https://www.awsarchitectureblog.com/2015/03/backoff.html">full jitter</a> strategy.
157      * The {@code minJitterRate} and {@code maxJitterRate} is used to calculate the lower and upper bound
158      * of the ultimate delay. The lower bound will be {@code ((1 - minJitterRate) * originalDelay)} and the
159      * upper bound will be {@code ((1 + maxJitterRate) * originalDelay)}. For example, if the delay
160      * returned by {@link Backoff#exponential(long, long)} is 1000 milliseconds and the {@code minJitterRate}
161      * is -0.2, {@code maxJitterRate} is 0.3, the ultimate backoff delay will be chosen between
162      * 1000 * (1 - 0.2) and 1000 * (1 + 0.3) by {@link ThreadLocalRandom}.
163      * The rate values should be between -1.0 and 1.0.
164      *
165      * @param minJitterRate the rate that used to calculate the lower bound of the backoff delay
166      * @param maxJitterRate the rate that used to calculate the upper bound of the backoff delay
167      * @throws IllegalArgumentException if {@code minJitterRate} is greater than {@code maxJitterRate} or if the
168      *                                  {@code minJitterRate} and {@code maxJitterRate} values are not in
169      *                                  between -1.0 and 1.0
170      */
171     default Backoff withJitter(double minJitterRate, double maxJitterRate) {
172         return withJitter(minJitterRate, maxJitterRate, ThreadLocalRandom::current);
173     }
174 
175     /**
176      * Returns a {@link Backoff} that adds a random jitter value to the original delay using
177      * <a href="https://www.awsarchitectureblog.com/2015/03/backoff.html">full jitter</a> strategy.
178      * The {@code minJitterRate} and {@code maxJitterRate} is used to calculate the lower and upper bound
179      * of the ultimate delay. The lower bound will be {@code ((1 - minJitterRate) * originalDelay)} and the
180      * upper bound will be {@code ((1 + maxJitterRate) * originalDelay)}. For example, if the delay
181      * returned by {@link Backoff#exponential(long, long)} is 1000 milliseconds and the {@code minJitterRate}
182      * is -0.2, {@code maxJitterRate} is 0.3, the ultimate backoff delay will be chosen between
183      * 1000 * (1 - 0.2) and 1000 * (1 + 0.3). The rate values should be between -1.0 and 1.0.
184      *
185      * @param minJitterRate  the rate that used to calculate the lower bound of the backoff delay
186      * @param maxJitterRate  the rate that used to calculate the upper bound of the backoff delay
187      * @param randomSupplier the supplier that provides {@link Random} in order to calculate the ultimate delay
188      * @throws IllegalArgumentException if {@code minJitterRate} is greater than {@code maxJitterRate} or if the
189      *                                  {@code minJitterRate} and {@code maxJitterRate} values are not in
190      *                                  between -1.0 and 1.0
191      */
192     default Backoff withJitter(double minJitterRate, double maxJitterRate, Supplier<Random> randomSupplier) {
193         return new JitterAddingBackoff(this, minJitterRate, maxJitterRate, randomSupplier);
194     }
195 
196     /**
197      * Returns a {@link Backoff} which limits the number of attempts up to the specified value.
198      */
199     default Backoff withMaxAttempts(int maxAttempts) {
200         return new AttemptLimitingBackoff(this, maxAttempts);
201     }
202 }
