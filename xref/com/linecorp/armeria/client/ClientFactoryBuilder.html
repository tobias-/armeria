57  /**
58   * Builds a new {@link ClientFactory}.
59   *
60   * <h2>Example</h2>
61   * <pre>{@code
62   * final ClientFactory factory = new ClientFactoryBuilder();
63   *     // Set the connection timeout to 5 seconds.
64   *     .connectTimeoutMillis(5000)
65   *     // Set the socket send buffer to 1 MiB.
66   *     .socketOption(ChannelOption.SO_SNDBUF, 1048576)
67   *     // Disable certificate verification; never do this in production!
68   *     .sslContextCustomizer(b -> b.trustManager(InsecureTrustManagerFactory.INSTANCE))
69   *     .build();
70   * }</pre>
71   */
72  public final class ClientFactoryBuilder {
73  
74      private static final KeyedChannelPoolHandler<PoolKey> DEFAULT_CONNECTION_POOL_LISTENER =
75              KeyedChannelPoolHandler.noop();
76  
77      private static final Consumer<SslContextBuilder> DEFAULT_SSL_CONTEXT_CUSTOMIZER = b -> { /* no-op */ };
78  
79      private static final Function<EventLoopGroup, DnsAddressResolverGroup>
80              DEFAULT_ADDRESS_RESOLVER_GROUP_FACTORY = eventLoopGroup -> new DnsAddressResolverGroup(
81              TransportType.datagramChannelType(eventLoopGroup),
82              DnsServerAddressStreamProviders.platformDefault());
83  
84      // Do not accept 1) the options that may break Armeria and 2) the deprecated options.
85      @SuppressWarnings("deprecation")
86      private static final Set<ChannelOption<?>> PROHIBITED_SOCKET_OPTIONS = ImmutableSet.of(
87              ChannelOption.ALLOW_HALF_CLOSURE, ChannelOption.AUTO_READ,
88              ChannelOption.AUTO_CLOSE, ChannelOption.MAX_MESSAGES_PER_READ,
89              ChannelOption.WRITE_BUFFER_HIGH_WATER_MARK, ChannelOption.WRITE_BUFFER_LOW_WATER_MARK,
90              EpollChannelOption.EPOLL_MODE);
91  
92      // Netty-related properties:
93      private EventLoopGroup workerGroup = CommonPools.workerGroup();
94      private boolean shutdownWorkerGroupOnClose;
95      private final Map<ChannelOption<?>, Object> socketOptions = new HashMap<>();
96      private Consumer<? super SslContextBuilder> sslContextCustomizer = DEFAULT_SSL_CONTEXT_CUSTOMIZER;
97      private Function<? super EventLoopGroup,
98              ? extends AddressResolverGroup<? extends InetSocketAddress>> addressResolverGroupFactory =
99              DEFAULT_ADDRESS_RESOLVER_GROUP_FACTORY;
100     private int initialHttp2ConnectionWindowSize = DEFAULT_WINDOW_SIZE;
101     private int initialHttp2StreamWindowSize = DEFAULT_WINDOW_SIZE;
102     private int http2MaxFrameSize = DEFAULT_MAX_FRAME_SIZE;
103     private int maxHttp1InitialLineLength = Flags.defaultMaxHttp1InitialLineLength();
104     private int maxHttp1HeaderSize = Flags.defaultMaxHttp1HeaderSize();
105     private int maxHttp1ChunkSize = Flags.defaultMaxHttp1ChunkSize();
106 
107     // Armeria-related properties:
108     private long idleTimeoutMillis = Flags.defaultClientIdleTimeoutMillis();
109     private boolean useHttp2Preface = Flags.defaultUseHttp2Preface();
110     private boolean useHttp1Pipelining = Flags.defaultUseHttp1Pipelining();
111     private KeyedChannelPoolHandler<? super PoolKey> connectionPoolListener = DEFAULT_CONNECTION_POOL_LISTENER;
112     private MeterRegistry meterRegistry = Metrics.globalRegistry;
113 
114     /**
115      * Creates a new instance.
116      */
117     public ClientFactoryBuilder() {
118         connectTimeoutMillis(Flags.defaultConnectTimeoutMillis());
119         socketOption(ChannelOption.SO_KEEPALIVE, true);
120     }
121 
122     /**
123      * Sets the worker {@link EventLoopGroup} which is responsible for performing socket I/O and running
124      * {@link Client#execute(ClientRequestContext, Request)}.
125      * If not set, {@linkplain CommonPools#workerGroup() the common worker group} is used.
126      *
127      * @param shutdownOnClose whether to shut down the worker {@link EventLoopGroup}
128      *                        when the {@link ClientFactory} is closed
129      */
130     public ClientFactoryBuilder workerGroup(EventLoopGroup workerGroup, boolean shutdownOnClose) {
131         this.workerGroup = requireNonNull(workerGroup, "workerGroup");
132         shutdownWorkerGroupOnClose = shutdownOnClose;
133         return this;
134     }
135 
136     /**
137      * Sets the timeout of a socket connection attempt.
138      */
139     public ClientFactoryBuilder connectTimeout(Duration connectTimeout) {
140         requireNonNull(connectTimeout, "connectTimeout");
141         checkArgument(!connectTimeout.isZero() && !connectTimeout.isNegative(),
142                       "connectTimeout: %s (expected: > 0)", connectTimeout);
143         return connectTimeoutMillis(connectTimeout.toMillis());
144     }
145 
146     /**
147      * Sets the timeout of a socket connection attempt in milliseconds.
148      */
149     public ClientFactoryBuilder connectTimeoutMillis(long connectTimeoutMillis) {
150         checkArgument(connectTimeoutMillis > 0,
151                       "connectTimeoutMillis: %s (expected: > 0)", connectTimeoutMillis);
152         return socketOption(ChannelOption.CONNECT_TIMEOUT_MILLIS,
153                             ConvertUtils.safeLongToInt(connectTimeoutMillis));
154     }
155 
156     /**
157      * Sets the options of sockets created by the {@link ClientFactory}.
158      */
159     public <T> ClientFactoryBuilder socketOption(ChannelOption<T> option, T value) {
160         requireNonNull(option, "option");
161         checkArgument(!PROHIBITED_SOCKET_OPTIONS.contains(option),
162                       "prohibited socket option: %s", option);
163 
164         socketOptions.put(option, requireNonNull(value, "value"));
165         return this;
166     }
167 
168     /**
169      * Sets the {@link Consumer} which can arbitrarily configure the {@link SslContextBuilder} that will be
170      * applied to the SSL session. For example, use {@link SslContextBuilder#trustManager} to configure a
171      * custom server CA or {@link SslContextBuilder#keyManager} to configure a client certificate for SSL
172      * authorization.
173      */
174     public ClientFactoryBuilder sslContextCustomizer(Consumer<? super SslContextBuilder> sslContextCustomizer) {
175         this.sslContextCustomizer = requireNonNull(sslContextCustomizer, "sslContextCustomizer");
176         return this;
177     }
178 
179     /**
180      * Sets the factory that creates a {@link AddressResolverGroup} which resolves remote addresses into
181      * {@link InetSocketAddress}es.
182      */
183     public ClientFactoryBuilder addressResolverGroupFactory(
184             Function<? super EventLoopGroup,
185                      ? extends AddressResolverGroup<? extends InetSocketAddress>> addressResolverGroupFactory) {
186         this.addressResolverGroupFactory = requireNonNull(addressResolverGroupFactory,
187                                                           "addressResolverGroupFactory");
188         return this;
189     }
190 
191     /**
192      * Sets the <a href="https://tools.ietf.org/html/rfc7540#section-6.9.2">initial connection flow-control
193      * window size</a>. The HTTP/2 connection is first established with
194      * {@value Http2CodecUtil#DEFAULT_WINDOW_SIZE} bytes of connection flow-control window size,
195      * and it is changed if and only if {@code initialHttp2ConnectionWindowSize} is set.
196      * Note that this setting affects the connection-level window size, not the window size of streams.
197      *
198      * @see #initialHttp2StreamWindowSize(int)
199      */
200     public ClientFactoryBuilder initialHttp2ConnectionWindowSize(int initialHttp2ConnectionWindowSize) {
201         checkArgument(initialHttp2ConnectionWindowSize >= DEFAULT_WINDOW_SIZE,
202                       "initialHttp2ConnectionWindowSize: %s (expected: >= %s and <= %s)",
203                       initialHttp2ConnectionWindowSize, DEFAULT_WINDOW_SIZE, MAX_INITIAL_WINDOW_SIZE);
204         this.initialHttp2ConnectionWindowSize = initialHttp2ConnectionWindowSize;
205         return this;
206     }
207 
208     /**
209      * Sets the <a href="https://tools.ietf.org/html/rfc7540#section-6.5.2">SETTINGS_INITIAL_WINDOW_SIZE</a>
210      * for HTTP/2 stream-level flow control. Note that this setting affects the window size of all streams,
211      * not the connection-level window size.
212      *
213      * @see #initialHttp2ConnectionWindowSize(int)
214      */
215     public ClientFactoryBuilder initialHttp2StreamWindowSize(int initialHttp2StreamWindowSize) {
216         checkArgument(initialHttp2StreamWindowSize > 0,
217                       "initialHttp2StreamWindowSize: %s (expected: > 0 and <= %s)",
218                       initialHttp2StreamWindowSize, MAX_INITIAL_WINDOW_SIZE);
219         this.initialHttp2StreamWindowSize = initialHttp2StreamWindowSize;
220         return this;
221     }
222 
223     /**
224      * Sets the <a href="https://tools.ietf.org/html/rfc7540#section-6.5.2">SETTINGS_MAX_FRAME_SIZE</a>
225      * that indicates the size of the largest frame payload that this client is willing to receive.
226      */
227     public ClientFactoryBuilder http2MaxFrameSize(int http2MaxFrameSize) {
228         checkArgument(http2MaxFrameSize >= MAX_FRAME_SIZE_LOWER_BOUND &&
229                       http2MaxFrameSize <= MAX_FRAME_SIZE_UPPER_BOUND,
230                       "http2MaxFramSize: %s (expected: >= %s and <= %s)",
231                       http2MaxFrameSize, MAX_FRAME_SIZE_LOWER_BOUND, MAX_FRAME_SIZE_UPPER_BOUND);
232         this.http2MaxFrameSize = http2MaxFrameSize;
233         return this;
234     }
235 
236     /**
237      * Sets the maximum length of an HTTP/1 response initial line.
238      */
239     public ClientFactoryBuilder maxHttp1InitialLineLength(int maxHttp1InitialLineLength) {
240         checkArgument(maxHttp1InitialLineLength >= 0,
241                       "maxHttp1InitialLineLength: %s (expected: >= 0)",
242                       maxHttp1InitialLineLength);
243         this.maxHttp1InitialLineLength = maxHttp1InitialLineLength;
244         return this;
245     }
246 
247     /**
248      * Sets the maximum length of all headers in an HTTP/1 response.
249      */
250     public ClientFactoryBuilder maxHttp1HeaderSize(int maxHttp1HeaderSize) {

258     /**
259      * Sets the maximum length of each chunk in an HTTP/1 response content.

262      */
263     public ClientFactoryBuilder maxHttp1ChunkSize(int maxHttp1ChunkSize) {

266                       maxHttp1ChunkSize);
267         this.maxHttp1ChunkSize = maxHttp1ChunkSize;
268         return this;
269     }

271     /**
272      * Sets the idle timeout of a socket connection. The connection is closed if there is no request in
273      * progress for this amount of time.
274      */
275     public ClientFactoryBuilder idleTimeout(Duration idleTimeout) {
276         requireNonNull(idleTimeout, "idleTimeout");
277         checkArgument(!idleTimeout.isNegative(), "idleTimeout: %s (expected: >= 0)", idleTimeout);
278         return idleTimeoutMillis(idleTimeout.toMillis());
279     }
280 
281     /**
282      * Sets the idle timeout of a socket connection in milliseconds. The connection is closed if there is no
283      * request in progress for this amount of time.
284      */
285     public ClientFactoryBuilder idleTimeoutMillis(long idleTimeoutMillis) {
286         checkArgument(idleTimeoutMillis >= 0, "idleTimeoutMillis: %s (expected: >= 0)", idleTimeoutMillis);
287         this.idleTimeoutMillis = idleTimeoutMillis;
288         return this;
289     }

291     /**
292      * Sets whether to send an HTTP/2 preface string instead of an HTTP/1 upgrade request to negotiate

295     public ClientFactoryBuilder useHttp2Preface(boolean useHttp2Preface) {

297         return this;
298     }
299 
300     /**
301      * Sets whether to use <a href="https://en.wikipedia.org/wiki/HTTP_pipelining">HTTP pipelining</a> for

304     public ClientFactoryBuilder useHttp1Pipelining(boolean useHttp1Pipelining) {
305         this.useHttp1Pipelining = useHttp1Pipelining;
306         return this;
307     }

309     /**
310      * Sets the listener which is notified on a connection pool event.
311      */
312     public ClientFactoryBuilder connectionPoolListener(

314         this.connectionPoolListener = requireNonNull(connectionPoolListener, "connectionPoolListener");
315         return this;

318     /**
319      * Sets the {@link MeterRegistry} which collects various stats.
320      */
321     public ClientFactoryBuilder meterRegistry(MeterRegistry meterRegistry) {
322         this.meterRegistry = requireNonNull(meterRegistry, "meterRegistry");
323         return this;

326     /**
327      * Returns a newly-created {@link ClientFactory} based on the properties of this builder.
328      */
329     public ClientFactory build() {
330         return new DefaultClientFactory(new HttpClientFactory(
331                 workerGroup, shutdownWorkerGroupOnClose, socketOptions, sslContextCustomizer,
332                 addressResolverGroupFactory, initialHttp2ConnectionWindowSize, initialHttp2StreamWindowSize,
333                 http2MaxFrameSize, maxHttp1InitialLineLength, maxHttp1HeaderSize,
334                 maxHttp1ChunkSize, idleTimeoutMillis, useHttp2Preface,
335                 useHttp1Pipelining, connectionPoolListener, meterRegistry));
336     }

338     @Override
339     public String toString() {
340         return toString(this, workerGroup, shutdownWorkerGroupOnClose, socketOptions,
341                         sslContextCustomizer, addressResolverGroupFactory, initialHttp2ConnectionWindowSize,
342                         initialHttp2StreamWindowSize, http2MaxFrameSize, maxHttp1InitialLineLength,
343                         maxHttp1HeaderSize, maxHttp1ChunkSize, idleTimeoutMillis,
344                         useHttp2Preface, useHttp1Pipelining, connectionPoolListener, meterRegistry);
345     }

347     static String toString(
348             ClientFactoryBuilder self,
349             EventLoopGroup workerGroup, boolean shutdownWorkerGroupOnClose,
350             Map<ChannelOption<?>, Object> socketOptions,
351             Consumer<? super SslContextBuilder> sslContextCustomizer,
352             Function<? super EventLoopGroup,
353                      ? extends AddressResolverGroup<? extends InetSocketAddress>> addressResolverGroupFactory,
354             int initialHttp2ConnectionWindowSize, int initialHttp2StreamWindowSize, int http2MaxFrameSize,
355             int maxHttp1InitialLineLength, int maxHttp1HeaderSize, int maxHttp1ChunkSize,
356             long idleTimeoutMillis, boolean useHttp2Preface,
357             boolean useHttp1Pipelining, KeyedChannelPoolHandler<? super PoolKey> connectionPoolListener,
358             MeterRegistry meterRegistry) {
359 
360         final ToStringHelper helper = MoreObjects.toStringHelper(self).omitNullValues();
361         helper.add("workerGroup", workerGroup + " (shutdownOnClose=" + shutdownWorkerGroupOnClose + ')')
362               .add("socketOptions", socketOptions)
363               .add("initialHttp2ConnectionWindowSize", initialHttp2ConnectionWindowSize)
364               .add("initialHttp2StreamWindowSize", initialHttp2StreamWindowSize)
365               .add("http2MaxFrameSize", http2MaxFrameSize)
366               .add("maxHttp1InitialLineLength", maxHttp1InitialLineLength)
367               .add("maxHttp1HeaderSize", maxHttp1HeaderSize)
368               .add("maxHttp1ChunkSize", maxHttp1ChunkSize)
369               .add("idleTimeoutMillis", idleTimeoutMillis)
370               .add("useHttp2Preface", useHttp2Preface)
371               .add("useHttp1Pipelining", useHttp1Pipelining);
372 
373         if (connectionPoolListener != DEFAULT_CONNECTION_POOL_LISTENER) {
374             helper.add("connectionPoolListener", connectionPoolListener);
375         }
376 
377         if (sslContextCustomizer != DEFAULT_SSL_CONTEXT_CUSTOMIZER) {
378             helper.add("sslContextCustomizer", sslContextCustomizer);
379         }
380 
381         if (addressResolverGroupFactory != DEFAULT_ADDRESS_RESOLVER_GROUP_FACTORY) {
382             helper.add("addressResolverGroupFactory", addressResolverGroupFactory);
383         }
384 
385         helper.add("meterRegistry", meterRegistry);
386 
387         return helper.toString();
388     }
