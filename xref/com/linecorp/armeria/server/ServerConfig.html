81      ServerConfig(
82              Iterable<ServerPort> ports,
83              VirtualHost defaultVirtualHost, Iterable<VirtualHost> virtualHosts,
84              EventLoopGroup workerGroup, boolean shutdownWorkerGroupOnStop,
85              int maxNumConnections, long idleTimeoutMillis,
86              long defaultRequestTimeoutMillis, long defaultMaxRequestLength,
87              int defaultMaxHttp1InitialLineLength, int defaultMaxHttp1HeaderSize, int defaultMaxHttp1ChunkSize,
88              Duration gracefulShutdownQuietPeriod, Duration gracefulShutdownTimeout,
89              Executor blockingTaskExecutor, MeterRegistry meterRegistry, String serviceLoggerPrefix,
90              Consumer<RequestLog> accessLogWriter) {
91  
92          requireNonNull(ports, "ports");
93          requireNonNull(defaultVirtualHost, "defaultVirtualHost");
94          requireNonNull(virtualHosts, "virtualHosts");
95  
96          // Set the primitive properties.
97          this.workerGroup = requireNonNull(workerGroup, "workerGroup");
98          this.shutdownWorkerGroupOnStop = shutdownWorkerGroupOnStop;
99          this.maxNumConnections = validateMaxNumConnections(maxNumConnections);
100         this.idleTimeoutMillis = validateIdleTimeoutMillis(idleTimeoutMillis);
101         this.defaultRequestTimeoutMillis = validateDefaultRequestTimeoutMillis(defaultRequestTimeoutMillis);
102         this.defaultMaxRequestLength = validateDefaultMaxRequestLength(defaultMaxRequestLength);
103         this.defaultMaxHttp1InitialLineLength = validateNonNegative(
104                 defaultMaxHttp1InitialLineLength, "defaultMaxHttp1InitialLineLength");
105         this.defaultMaxHttp1HeaderSize = validateNonNegative(
106                 defaultMaxHttp1HeaderSize, "defaultMaxHttp1HeaderSize");
107         this.defaultMaxHttp1ChunkSize = validateNonNegative(
108                 defaultMaxHttp1ChunkSize, "defaultMaxHttp1ChunkSize");
109         this.gracefulShutdownQuietPeriod = validateNonNegative(requireNonNull(
110                 gracefulShutdownQuietPeriod), "gracefulShutdownQuietPeriod");
111         this.gracefulShutdownTimeout = validateNonNegative(requireNonNull(
112                 gracefulShutdownTimeout), "gracefulShutdownTimeout");
113         validateGreaterThanOrEqual(gracefulShutdownTimeout, "gracefulShutdownTimeout",
114                                    gracefulShutdownQuietPeriod, "gracefulShutdownQuietPeriod");
115 
116         requireNonNull(blockingTaskExecutor, "blockingTaskExecutor");
117         if (blockingTaskExecutor instanceof ExecutorService) {
118             this.blockingTaskExecutor = new InterminableExecutorService((ExecutorService) blockingTaskExecutor);
119         } else {
120             this.blockingTaskExecutor = new ExecutorBasedExecutorService(blockingTaskExecutor);
121         }
122 
123         this.meterRegistry = requireNonNull(meterRegistry, "meterRegistry");
124         this.serviceLoggerPrefix = ServiceConfig.validateLoggerName(serviceLoggerPrefix, "serviceLoggerPrefix");
125         this.accessLogWriter = requireNonNull(accessLogWriter, "accessLogWriter");
126 
127         // Set localAddresses.
128         final List<ServerPort> portsCopy = new ArrayList<>();
129         for (ServerPort p : ports) {
130             if (p == null) {
131                 break;
132             }
133             portsCopy.add(p);
134         }
135 
136         if (portsCopy.isEmpty()) {
137             throw new IllegalArgumentException("no ports in the server");
138         }
139 
140         this.ports = Collections.unmodifiableList(portsCopy);
141 
142         // Set virtual host definitions and initialize their domain name mapping.
143         defaultVirtualHost = normalizeDefaultVirtualHost(defaultVirtualHost, portsCopy);
144         final DomainNameMappingBuilder<VirtualHost> mappingBuilder =
145                 new DomainNameMappingBuilder<>(defaultVirtualHost);
146         final List<VirtualHost> virtualHostsCopy = new ArrayList<>();
147         for (VirtualHost h : virtualHosts) {
148             if (h == null) {
149                 break;
150             }
151             virtualHostsCopy.add(h);
152             mappingBuilder.add(h.hostnamePattern(), h);
153         }
154         virtualHostMapping = mappingBuilder.build();
155 
156         // Add the default VirtualHost to the virtualHosts so that a user can retrieve all VirtualHosts
157         // via virtualHosts(). i.e. no need to check defaultVirtualHost().
158         virtualHostsCopy.add(defaultVirtualHost);
159 
160         // Sets the parent of VirtualHost to this configuration.
161         virtualHostsCopy.forEach(h -> h.setServerConfig(this));
162 
163         if (virtualHostsCopy.stream().allMatch(h -> h.serviceConfigs().isEmpty())) {
164             throw new IllegalArgumentException("no services in the server");
165         }
166 
167         this.virtualHosts = Collections.unmodifiableList(virtualHostsCopy);
168         this.defaultVirtualHost = defaultVirtualHost;
169 
170         // Build the complete list of the services available in this server.
171         services = virtualHostsCopy.stream()
172                                    .flatMap(h -> h.serviceConfigs().stream())
173                                    .collect(toImmutableList());
174     }

182             throw new IllegalArgumentException("idleTimeoutMillis: " + idleTimeoutMillis + " (expected: >= 0)");
183         }
184         return idleTimeoutMillis;

191         }

203     static int validateNonNegative(int value, String fieldName) {
204         if (value < 0) {
205             throw new IllegalArgumentException(fieldName + ": " + value + " (expected: >= 0)");
206         }
207         return value;
208     }

210     static Duration validateNonNegative(Duration duration, String fieldName) {
211         if (duration.isNegative()) {
212             throw new IllegalArgumentException(fieldName + ": " + duration + " (expected: >= 0)");
213         }
214         return duration;
215     }

217     static void validateGreaterThanOrEqual(Duration larger, String largerFieldName,
218                                            Duration smaller, String smallerFieldName) {
219         if (larger.compareTo(smaller) < 0) {
220             throw new IllegalArgumentException(largerFieldName + " must be greater than or equal to" +
221                                                smallerFieldName);
222         }
223     }

225     private static VirtualHost normalizeDefaultVirtualHost(VirtualHost h, List<ServerPort> ports) {
226         final SslContext sslCtx = h.sslContext();
227         // The default virtual host must have sslContext set if TLS is in use.
228         if (sslCtx == null && ports.stream().anyMatch(p -> p.protocol().isTls())) {
229             throw new IllegalArgumentException(
230                     "defaultVirtualHost must have sslContext set when TLS is enabled.");
231         }
232 
233         return new VirtualHost(
234                 h.defaultHostname(), "*", sslCtx,
235                 h.serviceConfigs().stream().map(
236                         e -> new ServiceConfig(e.pathMapping(), e.service(), e.loggerName().orElse(null)))
237                  .collect(Collectors.toList()), h.producibleMediaTypes());
238     }

295     /**
296      * Finds the {@link List} of {@link VirtualHost}s that contains the specified {@link Service}. If there's
297      * no match, an empty {@link List} is returned. Note that this is potentially an expensive operation and
298      * thus should not be used in a performance-sensitive path.
299      */
300     public List<VirtualHost> findVirtualHosts(Service<?, ?> service) {
301         requireNonNull(service, "service");
302 
303         @SuppressWarnings("rawtypes")
304         final Class<? extends Service> serviceType = service.getClass();
305         final List<VirtualHost> res = new ArrayList<>();
306         for (VirtualHost h : virtualHosts) {
307             for (ServiceConfig c : h.serviceConfigs()) {
308                 // Consider the case where the specified service is decorated before being added.
309                 final Service<?, ?> s = c.service();
310                 @SuppressWarnings("rawtypes")
311                 Optional<? extends Service> sOpt = s.as(serviceType);
312                 if (!sOpt.isPresent()) {
313                     continue;
314                 }
315 
316                 if (sOpt.get() == service) {
317                     res.add(c.virtualHost());
318                     break;
319                 }
320             }
321         }
322 
323         return res;
324     }

447     @Override
448     public String toString() {
449         String strVal = this.strVal;
450         if (strVal == null) {
451             this.strVal = strVal = toString(
452                     getClass(), ports(), null, virtualHosts(),
453                     workerGroup(), shutdownWorkerGroupOnStop(),
454                     maxNumConnections(), idleTimeoutMillis(),
455                     defaultRequestTimeoutMillis(), defaultMaxRequestLength(),
456                     defaultMaxHttp1InitialLineLength(), defaultMaxHttp1HeaderSize(), defaultMaxHttp1ChunkSize(),
457                     gracefulShutdownQuietPeriod(), gracefulShutdownTimeout(),
458                     blockingTaskExecutor(), meterRegistry(), serviceLoggerPrefix(), accessLogWriter());
459         }
460 
461         return strVal;
462     }

464     static String toString(
465             Class<?> type,
466             Iterable<ServerPort> ports, VirtualHost defaultVirtualHost, List<VirtualHost> virtualHosts,
467             EventLoopGroup workerGroup, boolean shutdownWorkerGroupOnStop,
468             int maxNumConnections, long idleTimeoutMillis, long defaultRequestTimeoutMillis,
469             long defaultMaxRequestLength, long defaultMaxHttp1InitialLineLength,
470             long defaultMaxHttp1HeaderSize, long defaultMaxHttp1ChunkSize,
471             Duration gracefulShutdownQuietPeriod, Duration gracefulShutdownTimeout,
472             Executor blockingTaskExecutor, MeterRegistry meterRegistry, String serviceLoggerPrefix,
473             Consumer<RequestLog> accessLogWriter) {
474 
475         StringBuilder buf = new StringBuilder();
476         if (type != null) {
477             buf.append(type.getSimpleName());
478         }
479 
480         buf.append("(ports: [");
481 
482         boolean hasPorts = false;
483         for (ServerPort p : ports) {
484             buf.append(ServerPort.toString(null, p.localAddress(), p.protocol()));
485             buf.append(", ");
486             hasPorts = true;
487         }
488 
489         if (hasPorts) {
490             buf.setCharAt(buf.length() - 2, ']');
491             buf.setCharAt(buf.length() - 1, ',');
492         } else {
493             buf.append("],");
494         }
495 
496         buf.append(" virtualHosts: [");
497         if (!virtualHosts.isEmpty()) {
498             virtualHosts.forEach(c -> {
499                 buf.append(VirtualHost.toString(null, c.defaultHostname(), c.hostnamePattern(),
500                                                 c.sslContext(), c.serviceConfigs()));
501                 buf.append(", ");
502             });
503 
504             if (defaultVirtualHost != null) {
505                 buf.append(VirtualHost.toString(null, defaultVirtualHost.defaultHostname(), "*",
506                                                 defaultVirtualHost.sslContext(),
507                                                 defaultVirtualHost.serviceConfigs()));
508             } else {
509                 buf.setLength(buf.length() - 2);
510             }
511         } else if (defaultVirtualHost != null) {
512             buf.append(VirtualHost.toString(null, defaultVirtualHost.defaultHostname(), "*",
513                                             defaultVirtualHost.sslContext(),
514                                             defaultVirtualHost.serviceConfigs()));
515         }
516 
517         buf.append("], workerGroup: ");
518         buf.append(workerGroup);
519         buf.append(" (shutdownOnStop=");
520         buf.append(shutdownWorkerGroupOnStop);
521         buf.append("), maxNumConnections: ");
522         buf.append(maxNumConnections);
523         buf.append(", idleTimeout: ");
524         buf.append(idleTimeoutMillis);
525         buf.append("ms, defaultRequestTimeout: ");
526         buf.append(defaultRequestTimeoutMillis);
527         buf.append("ms, defaultMaxRequestLength: ");
528         buf.append(defaultMaxRequestLength);
529         buf.append("B, defaultMaxHttp1InitialLineLength: ");
530         buf.append(defaultMaxHttp1InitialLineLength);
531         buf.append("B, defaultMaxHttp1HeaderSize: ");
532         buf.append(defaultMaxHttp1HeaderSize);
533         buf.append("B, defaultMaxHttp1ChunkSize: ");
534         buf.append(defaultMaxHttp1ChunkSize);
535         buf.append("B, gracefulShutdownQuietPeriod: ");
536         buf.append(gracefulShutdownQuietPeriod);
537         buf.append(", gracefulShutdownTimeout: ");
538         buf.append(gracefulShutdownTimeout);
539         buf.append(", blockingTaskExecutor: ");
540         buf.append(blockingTaskExecutor);
541         if (meterRegistry != null) {
542             buf.append(", meterRegistry: ");
543             buf.append(meterRegistry);
544         }
545         buf.append(", serviceLoggerPrefix: ");
546         buf.append(serviceLoggerPrefix);
547         buf.append(", accessLogWriter: ");
548         buf.append(accessLogWriter);
549         buf.append(')');
550 
551         return buf.toString();
552     }
