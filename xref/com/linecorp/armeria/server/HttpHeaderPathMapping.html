34  /**
35   * A {@link PathMapping} based on {@link HttpMethod} and {@link MediaType}.
36   */
37  final class HttpHeaderPathMapping implements PathMapping {
38  
39      private static final List<MediaType> ANY_TYPE = ImmutableList.of(MediaType.ANY_TYPE);
40      private static final Joiner loggerNameJoiner = Joiner.on('_');
41      private static final Joiner meterTagJoiner = Joiner.on(',');
42  
43      private final PathMapping pathStringMapping;
44      private final Set<HttpMethod> supportedMethods;
45      private final List<MediaType> consumeTypes;
46      private final List<MediaType> produceTypes;
47  
48      private final String loggerName;
49      private final String meterTag;
50  
51      private final int complexity;
52  
53      HttpHeaderPathMapping(PathMapping pathStringMapping, Set<HttpMethod> supportedMethods,
54                            List<MediaType> consumeTypes, List<MediaType> produceTypes) {
55          this.pathStringMapping = requireNonNull(pathStringMapping, "pathStringMapping");
56          this.supportedMethods = requireNonNull(supportedMethods, "supportedMethods");
57          this.consumeTypes = requireNonNull(consumeTypes, "consumeTypes");
58          this.produceTypes = requireNonNull(produceTypes, "produceTypes");
59  
60          loggerName = generateLoggerName(pathStringMapping.loggerName(),
61                                          supportedMethods, consumeTypes, produceTypes);
62          meterTag = generateMeterTag(pathStringMapping.meterTag(),
63                                      supportedMethods, consumeTypes, produceTypes);
64  
65          // Starts with 1 due to the HTTP method mapping.
66          int complexity = 1;
67          if (!consumeTypes.isEmpty()) {
68              complexity += 2;
69          }
70          if (!produceTypes.isEmpty()) {
71              complexity += 4;
72          }
73          this.complexity = complexity;
74      }
75  
76      @Override
77      public PathMappingResult apply(PathMappingContext mappingCtx) {
78          final PathMappingResult result = pathStringMapping.apply(mappingCtx);
79          if (!result.isPresent()) {
80              return PathMappingResult.empty();
81          }
82  
83          // We need to check the method at the last in order to return '405 Method Not Allowed'.
84          if (!supportedMethods.contains(mappingCtx.method())) {
85              // '415 Unsupported Media Type' and '406 Not Acceptable' is more specific than
86              // '405 Method Not Allowed'. So 405 would be set if there is no status code set before.
87              if (!mappingCtx.delayedThrowable().isPresent()) {
88                  mappingCtx.delayThrowable(HttpStatusException.of(HttpStatus.METHOD_NOT_ALLOWED));
89              }
90              return PathMappingResult.empty();
91          }
92  
93          if (!consumeTypes.isEmpty()) {
94              final MediaType type = mappingCtx.consumeType();
95              boolean found = false;
96              if (type != null) {
97                  for (MediaType mediaType : consumeTypes) {
98                      found = type.belongsTo(mediaType);
99                      if (found) {
100                         break;
101                     }
102                 }
103             }
104             if (!found) {
105                 mappingCtx.delayThrowable(HttpStatusException.of(HttpStatus.UNSUPPORTED_MEDIA_TYPE));
106                 return PathMappingResult.empty();
107             }
108         }
109 
110         // If a request does not have 'Accept' header, it would be handled the same as 'Accept: */*'.
111         // So there is a '*/*' at least in the 'mappingCtx.produceTypes()' if the virtual host supports
112         // the media type negotiation.
113         final List<MediaType> types = mappingCtx.produceTypes();
114         if (types == null) {
115             // Media type negotiation is not supported on this virtual host.
116             return result;
117         }
118 
119         // Also, the missing '@ProduceTypes' would be handled the same as '@ProduceTypes({"*/*"})'.
120         final List<MediaType> producibleTypes = produceTypes.isEmpty() ? ANY_TYPE : produceTypes;
121         for (MediaType produceType : producibleTypes) {
122             for (int i = 0; i < types.size(); i++) {
123                 if (produceType.belongsTo(types.get(i))) {
124                     // To early stop path mapping traversal,
125                     // we set the score as the best score when the index is 0.
126                     result.setScore(i == 0 ? PathMappingResult.HIGHEST_SCORE : -1 * i);
127                     if (!produceTypes.isEmpty()) {
128                         result.setNegotiatedProduceType(produceType);
129                     }
130                     return result;
131                 }
132             }
133         }
134 
135         mappingCtx.delayThrowable(HttpStatusException.of(HttpStatus.NOT_ACCEPTABLE));
136         return PathMappingResult.empty();
137     }

186     @Override
187     public String toString() {
188         final StringBuilder buf = new StringBuilder();
189         buf.append('[');
190         buf.append(pathStringMapping);
191         buf.append(", ");
192         buf.append(MoreObjects.toStringHelper("")
193                               .add("supportedMethods", supportedMethods)
194                               .add("consumeTypes", consumeTypes)
195                               .add("produceTypes", produceTypes));
196         buf.append(']');
197         return buf.toString();
198     }
199 
200     private static String generateLoggerName(String prefix, Set<HttpMethod> supportedMethods,
201                                              List<MediaType> consumeTypes, List<MediaType> produceTypes) {
202         final StringJoiner name = new StringJoiner(".");
203         name.add(prefix);
204         name.add(loggerNameJoiner.join(supportedMethods.stream().sorted().iterator()));
205 
206         // The following three cases should be different to each other.
207         // Each name would be produced as follows:
208         //
209         // consumeTypes: text/plain, text/html               -> consumes.text_plain.text_html
210         // consumeTypes: text/plain, produceTypes: text/html -> consumes.text_plain.produces.text_html
211         // produceTypes: text/plain, text/html               -> produces.text_plain.text_html
212 
213         if (!consumeTypes.isEmpty()) {
214             name.add("consumes");
215             consumeTypes.forEach(e -> name.add(e.type() + '_' + e.subtype()));
216         }
217         if (!produceTypes.isEmpty()) {
218             name.add("produces");
219             produceTypes.forEach(e -> name.add(e.type() + '_' + e.subtype()));
220         }
221         return name.toString();
222     }
223 
224     private static String generateMeterTag(String parentTag, Set<HttpMethod> supportedMethods,
225                                            List<MediaType> consumeTypes, List<MediaType> produceTypes) {
226 
227         final StringJoiner name = new StringJoiner(",");
228         name.add(parentTag);
229         name.add("methods:" + meterTagJoiner.join(supportedMethods.stream().sorted().iterator()));
230 
231         // The following three cases should be different to each other.
232         // Each name would be produced as follows:
233         //
234         // consumeTypes: text/plain, text/html               -> "consumes:text/plain,text/html"
235         // consumeTypes: text/plain, produceTypes: text/html -> "consumes:text/plain,produces:text/html"
236         // produceTypes: text/plain, text/html               -> "produces:text/plain,text/html"
237 
238         addMediaTypes(name, "consumes", consumeTypes);
239         addMediaTypes(name, "produces", produceTypes);
240 
241         return name.toString();
242     }
243 
244     private static void addMediaTypes(StringJoiner builder, String prefix, List<MediaType> mediaTypes) {
245         if (!mediaTypes.isEmpty()) {
246             final StringBuilder buf = new StringBuilder();
247             buf.append(prefix).append(':');
248             for (MediaType t : mediaTypes) {
249                 buf.append(t.type());
250                 buf.append('/');
251                 buf.append(t.subtype());
252                 buf.append(',');
253             }
254             buf.setLength(buf.length() - 1);
255             builder.add(buf.toString());
256         }
257     }
