43  /**
44   * A factory that creates a {@link Router} instance.
45   */
46  public final class Routers {
47      private static final Logger logger = LoggerFactory.getLogger(Routers.class);
48  
49      /**
50       * Returns the default implementation of the {@link Router} to find a {@link ServiceConfig}.
51       * It consists of several router implementations which use one of Trie and List. It also includes
52       * cache mechanism to improve its performance.
53       */
54      public static Router<ServiceConfig> ofVirtualHost(Iterable<ServiceConfig> configs) {
55          requireNonNull(configs, "configs");
56          return wrapVirtualHostRouter(defaultRouter(configs, ServiceConfig::pathMapping));
57      }
58  
59      /**
60       * Returns the default implementation of the {@link Router} to find a {@link CompositeServiceEntry}.
61       */
62      public static <I extends Request, O extends Response> Router<Service<I, O>> ofCompositeService(
63              List<CompositeServiceEntry<I, O>> entries) {
64          requireNonNull(entries, "entries");
65  
66          final Router<CompositeServiceEntry<I, O>> delegate =
67                  wrapCompositeServiceRouter(defaultRouter(entries, CompositeServiceEntry::pathMapping));
68          return new CompositeRouter<>(delegate, result ->
69                  result.isPresent() ? PathMapped.of(result.mapping(), result.mappingResult(),
70                                                     result.value().service())
71                                     : PathMapped.empty());
72      }
73  
74      /**
75       * Returns the default implementation of {@link Router}. It consists of several router implementations
76       * which use one of Trie and List. Consecutive {@link ServiceConfig}s would be grouped according to whether
77       * it is able to produce trie path string or not while traversing the list, then each group would be
78       * transformed to a {@link Router}.
79       */
80      private static <V> Router<V> defaultRouter(Iterable<V> values,
81                                                 Function<V, PathMapping> pathMappingResolver) {
82          return new CompositeRouter<>(routers(values, pathMappingResolver), Function.identity());
83      }
84  
85      /**
86       * Returns a list of {@link Router}s.
87       */
88      @VisibleForTesting
89      static <V> List<Router<V>> routers(Iterable<V> values, Function<V, PathMapping> pathMappingResolver) {
90          final ImmutableList.Builder<Router<V>> builder = ImmutableList.builder();
91          final List<V> group = new ArrayList<>();
92  
93          boolean addingTrie = true;
94  
95          for (V value : values) {
96              final PathMapping mapping = pathMappingResolver.apply(value);
97              final boolean triePathPresent = mapping.triePath().isPresent();
98              if (addingTrie && triePathPresent || !addingTrie && !triePathPresent) {
99                  // We are adding the same type of PathMapping to 'group'.
100                 group.add(value);
101                 continue;
102             }
103 
104             // Changed the router type.
105             if (!group.isEmpty()) {
106                 builder.add(router(addingTrie, group, pathMappingResolver));
107             }
108             addingTrie = !addingTrie;
109             group.add(value);
110         }
111         if (!group.isEmpty()) {
112             builder.add(router(addingTrie, group, pathMappingResolver));
113         }
114         return builder.build();
115     }
116 
117     /**
118      * Returns a {@link Router} implementation which is using one of {@link RoutingTrie} and {@link List}.
119      */
120     private static <V> Router<V> router(boolean isTrie, List<V> values,
121                                         Function<V, PathMapping> pathMappingResolver) {
122         final Comparator<V> valueComparator =
123                 Comparator.comparingInt(e -> -1 * pathMappingResolver.apply(e).complexity());
124 
125         final Router<V> router;
126         if (isTrie) {
127             final RoutingTrie.Builder<V> builder = new Builder<>();
128             // Set a comparator to sort services by the number of conditions to be checked in a descending
129             // order.
130             builder.comparator(valueComparator);
131             values.forEach(v -> builder.add(pathMappingResolver.apply(v).triePath().get(), v));
132             router = new TrieRouter<>(builder.build(), pathMappingResolver);
133         } else {
134             values.sort(valueComparator);
135             router = new SequentialRouter<>(values, pathMappingResolver);
136         }
137 
138         if (logger.isDebugEnabled()) {
139             logger.debug("Router created for {} service(s): {}",
140                          values.size(), router.getClass().getSimpleName());
141             values.forEach(c -> {
142                 final PathMapping mapping = pathMappingResolver.apply(c);
143                 logger.debug("meterTag: {}, complexity: {}", mapping.meterTag(), mapping.complexity());
144             });
145         }
146         values.clear();
147         return router;
148     }
149 
150     /**
151      * Finds the most suitable service from the given {@link ServiceConfig} list.
152      */
153     private static <V> PathMapped<V> findsBest(PathMappingContext mappingCtx, List<V> values,
154                                                Function<V, PathMapping> pathMappingResolver) {
155         PathMapped<V> result = PathMapped.empty();
156         if (values != null) {
157             for (V value : values) {
158                 final PathMapping mapping = pathMappingResolver.apply(value);
159                 final PathMappingResult mappingResult = mapping.apply(mappingCtx);
160                 if (mappingResult.isPresent()) {
161                     //
162                     // The services are sorted as follows:
163                     //
164                     // 1) annotated service with method and media type negotiation
165                     //    (consumable and producible)
166                     // 2) annotated service with method and producible media type negotiation
167                     // 3) annotated service with method and consumable media type negotiation
168                     // 4) annotated service with method negotiation
169                     // 5) the other services (in a registered order)
170                     //
171                     // 1) and 2) may produce a score between the lowest and the highest because they should
172                     // negotiate the produce type with the value of 'Accept' header.
173                     // 3), 4) and 5) always produces the lowest score.
174                     //
175 
176                     // Found the best matching.
177                     if (mappingResult.hasHighestScore()) {
178                         result = PathMapped.of(mapping, mappingResult, value);
179                         break;
180                     }
181 
182                     // This means that the 'mappingResult' is produced by one of 3), 4) and 5).
183                     // So we have no more chance to find a better matching from now.
184                     if (mappingResult.hasLowestScore()) {
185                         if (!result.isPresent()) {
186                             result = PathMapped.of(mapping, mappingResult, value);
187                         }
188                         break;
189                     }
190 
191                     // We have still a chance to find a better matching.
192                     if (result.isPresent()) {
193                         if (mappingResult.score() > result.mappingResult().score()) {
194                             // Replace the candidate with the new one only if the score is better.
195                             // If the score is same, we respect the order of service registration.
196                             result = PathMapped.of(mapping, mappingResult, value);
197                         }
198                     } else {
199                         // Keep the result as a candidate.
200                         result = PathMapped.of(mapping, mappingResult, value);
201                     }
202                 }
203             }
204         }
205         return result;
206     }

208     private static final class TrieRouter<V> implements Router<V> {
209 
210         private final RoutingTrie<V> trie;
211         private final Function<V, PathMapping> pathMappingResolver;
212 
213         TrieRouter(RoutingTrie<V> trie, Function<V, PathMapping> pathMappingResolver) {
214             this.trie = requireNonNull(trie, "trie");
215             this.pathMappingResolver = requireNonNull(pathMappingResolver, "pathMappingResolver");
216         }
217 
218         @Override
219         public PathMapped<V> find(PathMappingContext mappingCtx) {
220             return findsBest(mappingCtx, trie.find(mappingCtx.path()), pathMappingResolver);
221         }
222 
223         @Override
224         public void dump(OutputStream output) {
225             trie.dump(output);
226         }
227     }
228 
229     private static final class SequentialRouter<V> implements Router<V> {
230 
231         private final List<V> values;
232         private final Function<V, PathMapping> pathMappingResolver;
233 
234         SequentialRouter(List<V> values, Function<V, PathMapping> pathMappingResolver) {
235             this.values = ImmutableList.copyOf(requireNonNull(values, "values"));
236             this.pathMappingResolver = requireNonNull(pathMappingResolver, "pathMappingResolver");
237         }
238 
239         @Override
240         public PathMapped<V> find(PathMappingContext mappingCtx) {
241             return findsBest(mappingCtx, values, pathMappingResolver);
242         }
243 
244         @Override
245         public void dump(OutputStream output) {
246             // Do not close this writer in order to keep output stream open.
247             PrintWriter p = new PrintWriter(new OutputStreamWriter(output, StandardCharsets.UTF_8));
248             p.printf("Dump of %s:%n", this);
249             for (int i = 0; i < values.size(); i++) {
250                 p.printf("<%d> %s%n", i, values.get(i));
251             }
252             p.flush();
253         }
254     }
255
