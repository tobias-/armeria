97      private static final ChannelFutureListener CLOSE = future -> {
98          final Throwable cause = future.cause();
99          final Channel ch = future.channel();
100         if (cause != null) {
101             logException(ch, cause);
102         }
103         safeClose(ch);
104     };
105 
106     static final ChannelFutureListener CLOSE_ON_FAILURE = future -> {
107         final Throwable cause = future.cause();
108         if (cause != null && !(cause instanceof ClosedPublisherException)) {
109             final Channel ch = future.channel();
110             logException(ch, cause);
111             safeClose(ch);
112         }
113     };
114 
115     private static void logException(Channel ch, Throwable cause) {
116         final HttpServer server = HttpServer.get(ch);
117         if (server != null) {
118             Exceptions.logIfUnexpected(logger, ch, server.protocol(), cause);
119         } else {
120             Exceptions.logIfUnexpected(logger, ch, cause);
121         }
122     }
123 
124     static void safeClose(Channel ch) {
125         if (!ch.isActive()) {
126             return;
127         }
128 
129         // Do not call Channel.close() if AbstractHttp2ConnectionHandler.close() has been invoked
130         // already. Otherwise, it can trigger a bad cycle:
131         //
132         //   1. Channel.close() triggers AbstractHttp2ConnectionHandler.close().
133         //   2. AbstractHttp2ConnectionHandler.close() triggers Http2Stream.close().
134         //   3. Http2Stream.close() fails the promise of its pending writes.
135         //   4. The failed promise notifies this listener (CLOSE_ON_FAILURE).
136         //   5. This listener calls Channel.close().
137         //   6. Repeat from step 1.
138         //
139 
140         final AbstractHttp2ConnectionHandler h2handler =
141                 ch.pipeline().get(AbstractHttp2ConnectionHandler.class);
142 
143         if (h2handler == null || !h2handler.isClosing()) {
144             ch.close();
145         }
146     }

160     HttpServerHandler(ServerConfig config,
161                       GracefulShutdownSupport gracefulShutdownSupport,
162                       SessionProtocol protocol) {
163 
164         assert protocol == H1 || protocol == H1C || protocol == H2;
165 
166         this.config = requireNonNull(config, "config");
167         this.gracefulShutdownSupport = requireNonNull(gracefulShutdownSupport, "gracefulShutdownSupport");
168 
169         this.protocol = requireNonNull(protocol, "protocol");
170         if (protocol == H1 || protocol == H1C) {
171             responseEncoder = new Http1ObjectEncoder(true, protocol.isTls());
172         }
173 
174         accessLogWriter = config.accessLogWriter();
175     }

194     @Override
195     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
196         isReading = true; // Cleared in channelReadComplete()
197 
198         if (msg instanceof Http2Settings) {
199             handleHttp2Settings(ctx, (Http2Settings) msg);
200         } else {
201             handleRequest(ctx, (DecodedHttpRequest) msg);
202         }
203     }

205     private void handleHttp2Settings(ChannelHandlerContext ctx, Http2Settings h2settings) {
206         if (h2settings.isEmpty()) {
207             logger.trace("{} HTTP/2 settings: <empty>", ctx.channel());
208         } else {
209             logger.debug("{} HTTP/2 settings: {}", ctx.channel(), h2settings);
210         }
211 
212         if (protocol == H1) {
213             protocol = H2;
214         } else if (protocol == H1C) {
215             protocol = H2C;
216         }
217 
218         final Http2ConnectionHandler handler = ctx.pipeline().get(Http2ConnectionHandler.class);
219         if (responseEncoder == null) {
220             responseEncoder = new Http2ObjectEncoder(handler.encoder());
221         } else if (responseEncoder instanceof Http1ObjectEncoder) {
222             responseEncoder.close();
223             responseEncoder = new Http2ObjectEncoder(handler.encoder());
224         }
225     }

227     private void handleRequest(ChannelHandlerContext ctx, DecodedHttpRequest req) throws Exception {
228         // Ignore the request received after the last request,
229         // because we are going to close the connection after sending the last response.
230         if (handledLastRequest) {
231             return;
232         }
233 
234         // If we received the message with keep-alive disabled,
235         // we should not accept a request anymore.
236         if (!req.isKeepAlive()) {
237             handledLastRequest = true;
238         }
239 
240         final HttpHeaders headers = req.headers();
241         if (!ALLOWED_METHODS.contains(headers.method())) {
242             respond(ctx, req, HttpStatus.METHOD_NOT_ALLOWED,
243                     new IllegalArgumentException("Request method is not allowed: " +
244                                                  headers.method().name()));
245             return;
246         }
247 
248         // Handle 'OPTIONS * HTTP/1.1'.
249         final String originalPath = headers.path();
250         if (originalPath.isEmpty() || originalPath.charAt(0) != '/') {
251             if (headers.method() == HttpMethod.OPTIONS && "*".equals(originalPath)) {
252                 handleOptions(ctx, req);
253             } else {
254                 respond(ctx, req, HttpStatus.BAD_REQUEST,
255                         new IllegalArgumentException("Request path is invalid: " + originalPath));
256             }
257             return;
258         }
259 
260         // Validate and split path and query.
261         final PathAndQuery pathAndQuery = PathAndQuery.parse(originalPath);
262         if (pathAndQuery == null) {
263             // Reject requests without a valid path.
264             respond(ctx, req, HttpStatus.NOT_FOUND,
265                     new IllegalArgumentException("Request path is invalid: " + originalPath));
266             return;
267         }
268 
269         final String hostname = hostname(ctx, headers);
270         final VirtualHost host = config.findVirtualHost(hostname);
271 
272         final PathMappingContext mappingCtx = DefaultPathMappingContext.of(
273                 host, hostname, pathAndQuery.path(), pathAndQuery.query(), headers,
274                 host.producibleMediaTypes());
275         // Find the service that matches the path.
276         final PathMapped<ServiceConfig> mapped;
277         try {
278             mapped = host.findServiceConfig(mappingCtx);
279         } catch (HttpStatusException cause) {
280             // We do not need to handle HttpResponseException here because we do not use it internally.
281             respond(ctx, req, pathAndQuery, cause.httpStatus(), cause);
282             return;
283         } catch (Throwable cause) {
284             logger.warn("{} Unexpected exception: {}", ctx.channel(), req, cause);
285             respond(ctx, req, pathAndQuery, HttpStatus.INTERNAL_SERVER_ERROR, cause);
286             return;
287         }
288         if (!mapped.isPresent()) {
289             // No services matched the path.
290             handleNonExistentMapping(ctx, req, host, pathAndQuery, mappingCtx);
291             return;
292         }
293 
294         // Decode the request and create a new invocation context from it to perform an invocation.
295         final PathMappingResult mappingResult = mapped.mappingResult();
296         final ServiceConfig serviceCfg = mapped.value();
297         final Service<HttpRequest, HttpResponse> service = serviceCfg.service();
298 
299         final Channel channel = ctx.channel();
300         final DefaultServiceRequestContext reqCtx = new DefaultServiceRequestContext(
301                 serviceCfg, channel, serviceCfg.server().meterRegistry(),
302                 protocol, mappingCtx, mappingResult, req, getSSLSession(channel));
303 
304         try (SafeCloseable ignored = RequestContext.push(reqCtx)) {
305             final RequestLogBuilder logBuilder = reqCtx.logBuilder();
306             HttpResponse res;
307             try {
308                 req.init(reqCtx);
309                 res = service.serve(reqCtx, req);
310             } catch (HttpResponseException cause) {
311                 res = cause.httpResponse();
312             } catch (Throwable cause) {
313                 try {
314                     if (cause instanceof HttpStatusException) {
315                         respond(ctx, req, ((HttpStatusException) cause).httpStatus(), reqCtx, cause);
316                     } else {
317                         logger.warn("{} Unexpected exception: {}, {}", reqCtx, service, req, cause);
318                         respond(ctx, req, HttpStatus.INTERNAL_SERVER_ERROR, reqCtx, cause);
319                     }
320                 } finally {
321                     logBuilder.endRequest(cause);
322                     logBuilder.endResponse(cause);
323                 }
324                 return;
325             }
326 
327             final EventLoop eventLoop = channel.eventLoop();
328 
329             // Keep track of the number of unfinished requests and
330             // clean up the request stream when response stream ends.
331             gracefulShutdownSupport.inc();
332             unfinishedRequests++;
333 
334             if (pathAndQuery.query() == null && mapped.mapping().paramNames().isEmpty()) {
335                 reqCtx.log().addListener(log -> {
336                     HttpStatus status = log.responseHeaders().status();
337                     if (status != null && status.code() >= 200 && status.code() < 400) {
338                         pathAndQuery.storeInCache(originalPath);
339                     }
340                 }, RequestLogAvailability.COMPLETE);
341             }
342 
343             req.completionFuture().handle(voidFunction((ret, cause) -> {
344                 if (cause == null) {
345                     logBuilder.endRequest();
346                 } else {
347                     logBuilder.endRequest(cause);
348                 }
349             })).exceptionally(CompletionActions::log);
350 
351             res.completionFuture().handle(voidFunction((ret, cause) -> {
352                 req.abort();
353                 // NB: logBuilder.endResponse() is called by HttpResponseSubscriber below.
354                 eventLoop.execute(() -> {
355                     gracefulShutdownSupport.dec();
356                     if (--unfinishedRequests == 0 && handledLastRequest) {
357                         ctx.writeAndFlush(Unpooled.EMPTY_BUFFER).addListener(CLOSE);
358                     }
359                 });
360             })).exceptionally(CompletionActions::log);
361 
362             final HttpResponseSubscriber resSubscriber =
363                     new HttpResponseSubscriber(ctx, responseEncoder, reqCtx, req, accessLogWriter);
364             reqCtx.setRequestTimeoutChangeListener(resSubscriber);
365             res.subscribe(resSubscriber, eventLoop, true);
366         }
367     }

369     private void handleOptions(ChannelHandlerContext ctx, DecodedHttpRequest req) {
370         respond(ctx, req,
371                 AggregatedHttpMessage.of(
372                         HttpHeaders.of(HttpStatus.OK)
373                                    .set(HttpHeaderNames.ALLOW, ALLOWED_METHODS_STRING)),
374                 newEarlyRespondingRequestContext(ctx, req, req.path(), null), null);
375     }

377     private void handleNonExistentMapping(ChannelHandlerContext ctx, DecodedHttpRequest req,
378                                           VirtualHost host, PathAndQuery pathAndQuery,
379                                           PathMappingContext mappingCtx) {
380 
381         String path = mappingCtx.path();
382         if (path.charAt(path.length() - 1) != '/') {
383             // Handle the case where /path doesn't exist but /path/ exists.
384             final String pathWithSlash = path + '/';
385             if (host.findServiceConfig(mappingCtx.overridePath(pathWithSlash)).isPresent()) {
386                 final String location;
387                 final String originalPath = req.path();
388                 if (path.length() == originalPath.length()) {
389                     location = pathWithSlash;
390                 } else {
391                     location = pathWithSlash + originalPath.substring(path.length());
392                 }
393                 redirect(ctx, req, pathAndQuery, location);
394                 return;
395             }
396         }
397 
398         respond(ctx, req, HttpStatus.NOT_FOUND, null);
399     }

401     private String hostname(ChannelHandlerContext ctx, HttpHeaders headers) {
402         final String hostname = headers.authority();
403         if (hostname == null) {
404             // Fill the authority with the default host name and current port, just in case the client did not
405             // send it.
406             final String defaultHostname = config.defaultVirtualHost().defaultHostname();
407             final int port = ((InetSocketAddress) ctx.channel().localAddress()).getPort();
408             headers.authority(defaultHostname + ':' + port);
409             return defaultHostname;
410         }
411 
412         final int hostnameColonIdx = hostname.lastIndexOf(':');
413         if (hostnameColonIdx < 0) {
414             return hostname;
415         }
416 
417         return hostname.substring(0, hostnameColonIdx);
418     }

426                 newEarlyRespondingRequestContext(ctx, req, pathAndQuery.path(), pathAndQuery.query()),

442     private void respond(ChannelHandlerContext ctx, DecodedHttpRequest req, HttpStatus status,
443                          RequestContext reqCtx, @Nullable Throwable cause) {
444 
445         if (status.code() < 400) {
446             respond(ctx, req, AggregatedHttpMessage.of(HttpHeaders.of(status)), reqCtx, cause);
447             return;
448         }
449 
450         final HttpData content;
451         if (req.method() == HttpMethod.HEAD || ArmeriaHttpUtil.isContentAlwaysEmpty(status)) {
452             content = HttpData.EMPTY_DATA;
453         } else {
454             content = status.toHttpData();
455         }
456 
457         respond(ctx, req,
458                 AggregatedHttpMessage.of(
459                         HttpHeaders.of(status)
460                                    .contentType(ERROR_CONTENT_TYPE),
461                         content), reqCtx, cause);
462     }

464     private void respond(ChannelHandlerContext ctx, DecodedHttpRequest req, AggregatedHttpMessage res,
465                          RequestContext reqCtx, @Nullable Throwable cause) {
466         if (!handledLastRequest) {
467             addKeepAliveHeaders(req, res);
468             respond0(ctx, req, res, reqCtx, cause).addListener(CLOSE_ON_FAILURE);
469         } else {
470             // Note that it is perfectly fine not to set the 'content-length' header to the last response
471             // of an HTTP/1 connection. We set it anyway to work around overly strict HTTP clients that always
472             // require a 'content-length' header for non-chunked responses.
473             setContentLength(req, res);
474             respond0(ctx, req, res, reqCtx, cause).addListener(CLOSE);
475         }
476 
477         if (!isReading) {
478             ctx.flush();
479         }
480     }

482     private ChannelFuture respond0(ChannelHandlerContext ctx,
483                                    DecodedHttpRequest req, AggregatedHttpMessage res,
484                                    RequestContext reqCtx, @Nullable Throwable cause) {
485 
486         // No need to consume further since the response is ready.
487         req.close();
488 
489         final boolean trailingHeadersEmpty = res.trailingHeaders().isEmpty();
490         final boolean contentAndTrailingHeadersEmpty = res.content().isEmpty() && trailingHeadersEmpty;
491 
492         final RequestLogBuilder logBuilder = reqCtx.logBuilder();
493 
494         logBuilder.startResponse();
495         ChannelFuture future = responseEncoder.writeHeaders(
496                 ctx, req.id(), req.streamId(), res.headers(), contentAndTrailingHeadersEmpty);
497         logBuilder.responseHeaders(res.headers());
498         if (!contentAndTrailingHeadersEmpty) {
499             future = responseEncoder.writeData(
500                     ctx, req.id(), req.streamId(), res.content(), trailingHeadersEmpty);
501             logBuilder.increaseResponseLength(res.content().length());
502             if (!trailingHeadersEmpty) {
503                 future = responseEncoder.writeHeaders(
504                         ctx, req.id(), req.streamId(), res.trailingHeaders(), true);
505             }
506         }
507 
508         future.addListener(f -> {
509             if (cause == null && f.isSuccess()) {
510                 logBuilder.endResponse();
511             } else {
512                 // Respect the first specified cause.
513                 logBuilder.endResponse(firstNonNull(cause, f.cause()));
514             }
515             accessLogWriter.accept(reqCtx.log());
516         });
517         return future;
518     }

520     /**
521      * Sets the keep alive header as per:
522      * - https://www.w3.org/Protocols/HTTP/1.1/draft-ietf-http-v11-spec-01.html#Connection
523      */
524     private void addKeepAliveHeaders(HttpRequest req, AggregatedHttpMessage res) {
525         if (protocol == H1 || protocol == H1C) {
526             res.headers().set(HttpHeaderNames.CONNECTION, "keep-alive");
527         } else {
528             // Do not add the 'connection' header for HTTP/2 responses.
529             // See https://tools.ietf.org/html/rfc7540#section-8.1.2.2
530         }
531 
532         setContentLength(req, res);
533     }
534 
535     /**
536      * Sets the 'content-length' header to the response.
537      */
538     private static void setContentLength(HttpRequest req, AggregatedHttpMessage res) {
539         // https://www.w3.org/Protocols/rfc2616/rfc2616-sec4.html#sec4.4
540         // prohibits to send message body for below cases.
541         // and in those cases, content should be empty.
542         if (req.method() == HttpMethod.HEAD || ArmeriaHttpUtil.isContentAlwaysEmpty(res.status())) {
543             return;
544         }
545         res.headers().setInt(HttpHeaderNames.CONTENT_LENGTH, res.content().length());
546     }

548     private static SSLSession getSSLSession(Channel channel) {

578     private EarlyRespondingRequestContext newEarlyRespondingRequestContext(ChannelHandlerContext ctx,
579                                                                            DecodedHttpRequest req,
580                                                                            String path,
581                                                                            @Nullable String query) {
582         final Channel channel = ctx.channel();
583         final EarlyRespondingRequestContext reqCtx =
584                 new EarlyRespondingRequestContext(channel, NoopMeterRegistry.get(), protocol(),
585                                                   req.method(), path, query, req);
586         final String host = LoggingUtil.remoteHost(req.headers(), ctx.channel());
587 
588         final RequestLogBuilder logBuilder = reqCtx.logBuilder();
589         logBuilder.startRequest(channel, protocol(), host);
590         logBuilder.requestHeaders(req.headers());
591 
592         return reqCtx;
593     }

595     private static final class EarlyRespondingRequestContext extends NonWrappingRequestContext {
596 
597         private final Channel channel;
598         private final DefaultRequestLog requestLog;
599 
600         EarlyRespondingRequestContext(Channel channel, MeterRegistry meterRegistry,
601                                       SessionProtocol sessionProtocol, HttpMethod method, String path,
602                                       @Nullable String query, Request request) {
603             super(meterRegistry, sessionProtocol, method, path, query, request);
604             this.channel = requireNonNull(channel, "channel");
605             requestLog = new DefaultRequestLog(this);
606         }
607 
608         @Override
609         public RequestContext newDerivedContext() {
610             return newDerivedContext(request());
611         }
612 
613         @Override
614         public RequestContext newDerivedContext(Request request) {
615             // There are no attributes which should be copied to a new instance.
616             return new EarlyRespondingRequestContext(channel(), meterRegistry(), sessionProtocol(),
617                                                      method(), path(), query(), request);
618         }
619 
620         @Nullable
621         @Override
622         protected Channel channel() {
623             return channel;
624         }
625 
626         @Override
627         public RequestLog log() {
628             return requestLog;
629         }
630 
631         @Override
632         public RequestLogBuilder logBuilder() {
633             return requestLog;
634         }
635 
636         @Override
637         public EventLoop eventLoop() {
638             return channel.eventLoop();
639         }
