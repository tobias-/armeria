171     static {
172         if (!Epoll.isAvailable()) {
173             final Throwable cause = Exceptions.peel(Epoll.unavailabilityCause());
174             logger.info("/dev/epoll not available: {}", cause.toString());
175         } else if (USE_EPOLL) {
176             logger.info("Using /dev/epoll");
177         }
178 
179         if (!OpenSsl.isAvailable()) {
180             final Throwable cause = Exceptions.peel(OpenSsl.unavailabilityCause());
181             logger.info("OpenSSL not available: {}", cause.toString());
182         } else if (USE_OPENSSL) {
183             logger.info("Using OpenSSL: {}, 0x{}",
184                         OpenSsl.versionString(),
185                         Long.toHexString(OpenSsl.version() & 0xFFFFFFFFL));
186         }
187     }

457     private static Optional<String> caffeineSpec(String name, String defaultValue) {
458         final String spec = get(name, defaultValue, value -> {
459             try {
460                 if (!"off".equals(value)) {
461                     CaffeineSpec.parse(value);
462                 }
463                 return true;
464             } catch (Exception e) {
465                 return false;
466             }
467         });
468         return "off".equals(spec) ? Optional.empty()
469                                   : Optional.of(spec);
470     }

476     private static boolean getBoolean(String name, boolean defaultValue, Predicate<Boolean> validator) {
477         return "true".equals(getNormalized(name, String.valueOf(defaultValue), value -> {
478             if ("true".equals(value)) {
479                 return validator.test(true);
480             }
481 
482             if ("false".equals(value)) {
483                 return validator.test(false);
484             }
485 
486             return false;
487         }));
488     }

490     private static int getInt(String name, int defaultValue, IntPredicate validator) {
491         return Integer.parseInt(getNormalized(name, String.valueOf(defaultValue), value -> {
492             try {
493                 return validator.test(Integer.parseInt(value));
494             } catch (Exception e) {
495                 // null or non-integer
496                 return false;
497             }
498         }));
499     }

501     private static long getLong(String name, long defaultValue, LongPredicate validator) {

505             } catch (Exception e) {

507                 return false;
508             }
509         }));

512     private static String get(String name, String defaultValue, Predicate<String> validator) {
513         final String fullName = PREFIX + name;
514         final String value = System.getProperty(fullName);
515         if (value == null) {
516             logger.info("{}: {} (default)", fullName, defaultValue);
517             return defaultValue;
518         }
519 
520         if (validator.test(value)) {
521             logger.info("{}: {}", fullName, value);
522             return value;
523         }
524 
525         logger.info("{}: {} (default instead of: {})", fullName, defaultValue, value);
526         return defaultValue;
527     }
528 
529     private static String getNormalized(String name, String defaultValue, Predicate<String> validator) {
530         final String fullName = PREFIX + name;
531         final String value = getLowerCased(fullName);
532         if (value == null) {
533             logger.info("{}: {} (default)", fullName, defaultValue);
534             return defaultValue;
535         }
536 
537         if (validator.test(value)) {
538             logger.info("{}: {}", fullName, value);
539             return value;
540         }
541 
542         logger.info("{}: {} (default instead of: {})", fullName, defaultValue, value);
543         return defaultValue;
544     }
545 
546     private static String getLowerCased(String fullName) {
547         String value = System.getProperty(fullName);
548         if (value != null) {
549             value = Ascii.toLowerCase(value);
550         }
551         return value;
552     }
