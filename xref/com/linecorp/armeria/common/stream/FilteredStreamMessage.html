17  package com.linecorp.armeria.common.stream;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.concurrent.CompletableFuture;
22  
23  import org.reactivestreams.Subscriber;
24  import org.reactivestreams.Subscription;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import com.linecorp.armeria.internal.PooledObjects;
29  
30  import io.netty.buffer.ByteBuf;
31  import io.netty.buffer.ByteBufHolder;
32  import io.netty.util.ReferenceCountUtil;
33  import io.netty.util.concurrent.EventExecutor;
34  
35  /**
36   * A {@link StreamMessage} that filters objects as they are published. The filtering
37   * will happen from an I/O thread, meaning the order of the filtering will match the
38   * order that the {@code delegate} processes the objects in.
39   */
40  public abstract class FilteredStreamMessage<T, U> implements StreamMessage<U> {
41  
42      private static final Logger logger = LoggerFactory.getLogger(FilteredStreamMessage.class);
43  
44      private final StreamMessage<T> delegate;
45      private final boolean withPooledObjects;
46  
47      /**
48       * Creates a new {@link FilteredStreamMessage} that filters objects published by {@code delegate}
49       * before passing to a subscriber.
50       */
51      protected FilteredStreamMessage(StreamMessage<T> delegate) {
52          this(delegate, false);
53      }
54  
55      /**
56       * Creates a new {@link FilteredStreamMessage} that filters objects published by {@code delegate}
57       * before passing to a subscriber.
58       *
59       * @param withPooledObjects if {@code true}, {@link #filter(Object)} receives the pooled {@link ByteBuf}
60       *                          and {@link ByteBufHolder} as is, without making a copy. If you don't know what
61       *                          this means, use {@link #FilteredStreamMessage(StreamMessage)}.
62       */
63      protected FilteredStreamMessage(StreamMessage<T> delegate, boolean withPooledObjects) {
64          requireNonNull(delegate, "delegate");
65          this.delegate = delegate;
66          this.withPooledObjects = withPooledObjects;
67      }
68  
69      /**
70       * The filter to apply to published objects. The result of the filter is passed on
71       * to the delegate subscriber.
72       */
73      protected abstract U filter(T obj);
74  
75      /**
76       * A callback executed just before calling {@link Subscriber#onSubscribe(Subscription)} on
77       * {@code subscriber}. Override this method to execute any initialization logic that may be needed.
78       */
79      protected void beforeSubscribe(Subscriber<? super U> subscriber, Subscription subscription) {}
80  
81      /**
82       * A callback executed just before calling {@link Subscriber#onComplete()} on {@code subscriber}.
83       * Override this method to execute any cleanup logic that may be needed before completing the
84       * subscription.
85       */
86      protected void beforeComplete(Subscriber<? super U> subscriber) {}
87  
88      /**
89       * A callback executed just before calling {@link Subscriber#onError(Throwable)} on {@code subscriber}.
90       * Override this method to execute any cleanup logic that may be needed before failing the
91       * subscription. This method may rewrite the {@code cause} and then return a new one so that the new
92       * {@link Throwable} would be passed to {@link Subscriber#onError(Throwable)}.
93       */
94      protected Throwable beforeError(Subscriber<? super U> subscriber, Throwable cause) {
95          return cause;
96      }
97  
98      @Override
99      public boolean isOpen() {
100         return delegate.isOpen();
101     }
102 
103     @Override
104     public boolean isEmpty() {
105         return delegate.isEmpty();
106     }
107 
108     @Override
109     public CompletableFuture<Void> completionFuture() {
110         return delegate.completionFuture();
111     }
112 
113     @Override
114     public void subscribe(Subscriber<? super U> subscriber) {
115         requireNonNull(subscriber, "subscriber");
116         delegate.subscribe(new FilteringSubscriber(subscriber));
117     }
118 
119     @Override
120     public void subscribe(Subscriber<? super U> subscriber, boolean withPooledObjects) {
121         requireNonNull(subscriber, "subscriber");
122         delegate.subscribe(new FilteringSubscriber(subscriber), withPooledObjects);
123     }
124 
125     @Override
126     public void subscribe(Subscriber<? super U> subscriber, EventExecutor executor) {
127         requireNonNull(subscriber, "subscriber");
128         requireNonNull(executor, "executor");
129         delegate.subscribe(new FilteringSubscriber(subscriber), executor);
130     }
131 
132     @Override
133     public void subscribe(Subscriber<? super U> subscriber, EventExecutor executor,
134                           boolean withPooledObjects) {
135         requireNonNull(subscriber, "subscriber");
136         requireNonNull(executor, "executor");
137         delegate.subscribe(new FilteringSubscriber(subscriber), executor, withPooledObjects);
138     }
139 
140     @Override
141     public void abort() {
142         delegate.abort();
143     }
144 
145     private final class FilteringSubscriber implements Subscriber<T> {
146 
147         private final Subscriber<? super U> delegate;
148 
149         FilteringSubscriber(Subscriber<? super U> delegate) {
150             requireNonNull(delegate, "delegate");
151             this.delegate = delegate;
152         }
153 
154         @Override
155         public void onSubscribe(Subscription s) {
156             beforeSubscribe(delegate, s);
157             delegate.onSubscribe(s);
158         }
159 
160         @Override
161         public void onNext(T o) {
162             ReferenceCountUtil.touch(o);
163             if (!withPooledObjects) {
164                 o = PooledObjects.toUnpooled(o);
165             }
166             delegate.onNext(filter(o));
167         }
168 
169         @Override
170         public void onError(Throwable t) {
171             final Throwable filteredCause = beforeError(delegate, t);
172             if (filteredCause != null) {
173                 delegate.onError(filteredCause);
174             } else {
175                 logger.warn("{}#beforeError() returned null. Using the original exception:",
176                             FilteredStreamMessage.this.getClass().getName(), t.toString());
177                 delegate.onError(t);
178             }
179         }
180 
181         @Override
182         public void onComplete() {
183             beforeComplete(delegate);
184             delegate.onComplete();
185         }
186     }
187 }
